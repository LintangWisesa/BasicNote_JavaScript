
// Operator precedent : prioritas x / + -


var usiaAndi = 40;
var usiaBudi = 20;

console.log(usiaAndi * usiaBudi); 
console.log(usiaAndi / usiaBudi);
console.log(usiaAndi + usiaBudi);
console.log(usiaAndi - usiaBudi);

console.log(usiaAndi++);     //usiaAndi++ = usiaAndi + 1
console.log(usiaAndi--);     //usiaAndi-- = usiaAndi - 1

console.log(usiaAndi *= 2);  //usiaAndi = usiaAndi * 2
console.log(usiaAndi /= 2);  //usiaAndi = usiaAndi / 2
console.log(usiaAndi += 2);  //usiaAndi = usiaAndi + 2
console.log(usiaAndi -= 2);  //usiaAndi = usiaAndi - 2
console.log(usiaAndi %= 2);
